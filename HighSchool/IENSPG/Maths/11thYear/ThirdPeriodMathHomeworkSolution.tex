\section*{ACTIVIDAD 1}

Halla los primeros cinco términos de cada sucesión

\begin{enumerate}[A.]
    \item $a_n = \frac{n + 1}{n}$
    \item $a_n = -\frac{1}{2}n $
    \item $a_n = \frac{n + 1}{n^2}$
\end{enumerate}

\subsection*{A.}

\[ a_n = \frac{n + 1}{n} \]

\[ a_1 = \frac{1 + 1}{1} \]
\[ a_1 = \frac{2}{1} \]
\[ a_1 = 2 \]


\[ a_2 = \frac{2 + 1}{2} \]
\[ a_2 = \frac{3}{2} \]

\[ a_3 = \frac{3 + 1}{3} \]
\[ a_3 = \frac{4}{3} \]

\[ a_4 = \frac{4 + 1}{4} \]
\[ a_4 = \frac{5}{4} \]

\[ a_5 = \frac{5 + 1}{5} \]
\[ a_5 = \frac{6}{5} \]

\subsection*{B.}

\[ a_n = -\frac{1}{2}n \]

\[ a_1 = -\frac{1}{2}1 \]
\[ a_1 = -\frac{1}{2} \]

\[ a_2 = -\frac{1}{2}2 \]
\[ a_2 = -1 \]

\[ a_3 = -\frac{1}{2}3 \]
\[ a_3 = -\frac{3}{2} \]

\[ a_4 = -\frac{1}{2}4 \]
\[ a_4 = -2 \]

\[ a_5 = -\frac{1}{2}5 \]
\[ a_5 = -\frac{5}{2} \]


\subsection*{C.}

\[ a_n = \frac{n + 1}{n^2} \]

\[ a_1 = \frac{1 + 1}{1^2} \]
\[ a_1 = \frac{2}{1} \]
\[ a_1 = 2 \]


\[ a_2 = \frac{2 + 1}{2^2} \]
\[ a_2 = \frac{3}{4} \]

\[ a_3 = \frac{3 + 1}{3^2} \]
\[ a_3 = \frac{4}{9} \]

\[ a_4 = \frac{4 + 1}{4^2} \]
\[ a_4 = \frac{5}{16} \]

\[ a_5 = \frac{5 + 1}{5^2} \]
\[ a_5 = \frac{6}{25} \]

    \item $a_n = \frac{n}{3n + 1}$
    \item $a_n = \frac{5n^2 + 1}{n^2 - 2n + 3}$
    \item $a_n = \frac{50n + 20}{n^2 - 1}$
\end{enumerate}

\subsection*{A.}
